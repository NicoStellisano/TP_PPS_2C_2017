  descargarArchivo(){
    let miAula:string;

    this.listaAlumnoItem.forEach(aula => {
      console.log(aula);
      if(aula.aula == this.aula){
        miAula = aula.aula;
      }
    });

    if(miAula == this.aula){
      //alert("Ir a descarga");
      this.navCtrl.push(DescargarArchivoPage,{aulaa:this.aula});
    }else{
      //alert("No hya nada que descargar");

  }
